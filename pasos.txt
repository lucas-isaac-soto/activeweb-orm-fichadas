empleado tendra como atributos:
    id
    nombre 
    apellido
    documento
    id_ciudad           ->   tabla ciudad
    id_sucursal         ->   tabla sucursal

    sueldo

    id 
    nombre

sucursal tendra como atributos:
    id
    id_ciudad
    id_empleado_gerente

movimientos tendra como atributos
    id
    id_empleado
    fecha_hora_ingreso
